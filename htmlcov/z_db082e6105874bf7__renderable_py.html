622 def seek(self, offset: int, whence: Seek = Seek.START) -> int: 

623 """Sets the current frame number. 

624 

625 Args: 

626 offset: Frame offset (relative to *whence*). 

627 whence: Reference position for *offset*. 

628 

629 Returns: 

630 The new current frame number. 

631 

632 Raises: 

633 IndefiniteSeekError: The renderable has 

634 :py:attr:`~term_image.renderable.FrameCount.INDEFINITE` frame count. 

635 ValueError: *offset* is out of range. 

636 

637 The value range for *offset* depends on *whence*: 

638 

639 .. list-table:: 

640 :align: left 

641 :header-rows: 1 

642 :widths: auto 

643 

644 * - *whence* 

645 - Valid value range for *offset* 

646 

647 * - :py:attr:`~term_image.renderable.Seek.START` 

648 - ``0`` <= *offset* < :py:attr:`frame_count` 

649 * - :py:attr:`~term_image.renderable.Seek.CURRENT` 

650 - -:py:meth:`tell` <= *offset* < :py:attr:`frame_count` - :py:meth:`tell` 

651 * - :py:attr:`~term_image.renderable.Seek.END` 

652 - -:py:attr:`frame_count` < *offset* <= ``0`` 

653 """ 

654 frame_count = self.frame_count 

655 

656 if frame_count is FrameCount.INDEFINITE: 

657 raise IndefiniteSeekError( 

658 "Cannot seek a renderable with INDEFINITE frame count" 

659 ) 

660 

661 frame = ( 

662 offset 

663 if whence is Seek.START 

664 else ( 

665 self._frame + offset 

666 if whence is Seek.CURRENT 

667 else frame_count + offset - 1 

668 ) 

669 ) 

670 if not 0 <= frame < frame_count: 

671 raise arg_value_error_range( 

672 "offset", 

673 offset, 

674 ( 

675 f"whence={whence.name}, frame_count={frame_count}" 

676 + (f", current={self._frame}" if whence is Seek.CURRENT else "") 

677 ), 

678 ) 

679 self._frame = frame 

680 

681 return frame

695 def _animate_( 

696 self, 

697 render_data: RenderData, 

698 render_args: RenderArgs, 

699 padding: Padding, 

700 loops: int, 

701 cache: bool | int, 

702 output: TextIO, 

703 ) -> None: 

704 """Animates frames of a renderable. 

705 

706 Args: 

707 render_data: Render data. 

708 render_args: Render arguments associated with the renderable's class. 

709 output: The text I/O stream to which rendered frames will be written. 

710 

711 All other parameters are the same as for :py:meth:`draw`, except that 

712 *padding* must have **absolute** dimensions if it's an instance of 

713 :py:class:`~term_image.padding.AlignedPadding`. 

714 

715 This is called by :py:meth:`draw` for animations. 

716 

717 NOTE: 

718 * The base implementation does not finalize *render_data*. 

719 * :term:`Render size` validation is expected to have been performed by 

720 the caller. 

721 * When called by :py:meth:`draw` (at least, the base implementation), 

722 *loops* and *cache* wouldn't have been validated. 

723 """ 

724 from term_image.render import RenderIterator 

725 

726 render_size: Size = render_data[Renderable].size 

727 height = render_size.height 

728 pad_left, _, _, pad_bottom = padding._get_exact_dimensions_(render_size) 

729 render_iter = RenderIterator._from_render_data_( 

730 self, 

731 render_data, 

732 render_args, 

733 padding, 

734 loops, 

735 False if loops == 1 else cache, 

736 finalize=False, 

737 ) 

738 cursor_to_bottom = cursor_down(height + pad_bottom - 1) 

739 cursor_to_next_render_line = f"\n{cursor_forward(pad_left)}" 

740 cursor_to_render_top_left = ( 

741 f"\r{cursor_up(height - 1)}{cursor_forward(pad_left)}" 

742 ) 

743 write = output.write 

744 flush = output.flush 

745 

746 try: 

747 # first frame 

748 try: 

749 frame = next(render_iter) 

750 except StopIteration: # `INDEFINITE` frame count 

751 return 

752 

753 try: 

754 write(frame.render_output) 

755 flush() 

756 except KeyboardInterrupt: 

757 self._handle_interrupted_draw_(render_data, render_args, output) 

758 return 

759 else: 

760 write( 

761 f"\r{cursor_up(height + pad_bottom - 1)}{cursor_forward(pad_left)}" 

762 ) 

763 flush() 

764 

765 # Padding has been drawn with the first frame, only the actual render is 

766 # needed henceforth. 

767 render_iter.set_padding(ExactPadding()) 

768 

769 # render next frame during previous frame's duration 

770 duration_ms = frame.duration 

771 start_ns = perf_counter_ns() 

772 

773 for frame in render_iter: # Render next frame 

774 # left-over of previous frame's duration 

775 sleep( 

776 max(0, duration_ms * 10**6 - (perf_counter_ns() - start_ns)) / 10**9 

777 ) 

778 

779 # clear previous frame, if necessary 

780 self._clear_frame_(render_data, render_args, pad_left + 1, output) 

781 

782 # draw next frame 

783 try: 

784 write(frame.render_output.replace("\n", cursor_to_next_render_line)) 

785 flush() 

786 except KeyboardInterrupt: 

787 self._handle_interrupted_draw_(render_data, render_args, output) 

788 return 

789 

790 write(cursor_to_render_top_left) 

791 flush() 

792 

793 # render next frame during previous frame's duration 

794 start_ns = perf_counter_ns() 

795 duration_ms = frame.duration 

796 

797 # left-over of last frame's duration 

798 sleep(max(0, duration_ms * 10**6 - (perf_counter_ns() - start_ns)) / 10**9) 

799 except KeyboardInterrupt: 

800 pass 

801 finally: 

802 render_iter.close() 

803 # Move the cursor to the last line to prevent "overlaid" output in a 

804 # terminal 

805 write(cursor_to_bottom) 

806 flush()
